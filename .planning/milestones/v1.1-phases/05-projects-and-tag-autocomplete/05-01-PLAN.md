export const projects = sqliteTable('projects', {
  id: integer('id').primaryKey({ autoIncrement: true }),

CREATE TABLE IF NOT EXISTS `project_bullets` (
  `id` integer PRIMARY KEY AUTOINCREMENT NOT NULL,
  `project_id` integer NOT NULL,
  `text` text NOT NULL,
  `sort_order` integer DEFAULT 0 NOT NULL,
  FOREIGN KEY (`project_id`) REFERENCES `projects`(`id`) ON DELETE cascade

**Handlers (src/main/handlers/projects.ts):** Create `registerProjectHandlers()` with these IPC channels, mirroring jobs.ts + bullets.ts:
- `projects:list` — select all projects ordered by `asc(projects.id)` (insertion order for v1.1), for each project load bullets ordered by `asc(projectBullets.sortOrder)`, return array of `{...project, bullets}`
- `projects:create` — insert with `{name, sortOrder: 0}`, return the new row
- `projects:update` — update by id with `{name?}`, return updated row
- `projects:delete` — delete by id (bullets cascade)
- `projectBullets:create` — insert with `{projectId, text, sortOrder}`, return new row
- `projectBullets:update` — update by id with `{text?}`, return updated row
- `projectBullets:delete` — delete by id
- `projectBullets:reorder` — loop over orderedIds, set sortOrder = index

**Handler registration (src/main/handlers/index.ts):** Import `registerProjectHandlers` from `./projects` and call it in `registerAllHandlers()`.

**Preload bridge (src/preload/index.ts):** Add `projects` and `projectBullets` namespaces to the `api` object:
```typescript
projects: {
  list: () => ipcRenderer.invoke('projects:list'),
  create: (data: { name: string }) => ipcRenderer.invoke('projects:create', data),
  update: (id: number, data: { name?: string }) => ipcRenderer.invoke('projects:update', id, data),
  delete: (id: number) => ipcRenderer.invoke('projects:delete', id),
},
projectBullets: {
  create: (data: { projectId: number; text: string; sortOrder: number }) =>
    ipcRenderer.invoke('projectBullets:create', data),
  update: (id: number, data: { text?: string }) => ipcRenderer.invoke('projectBullets:update', id, data),
  delete: (id: number) => ipcRenderer.invoke('projectBullets:delete', id),
  reorder: (projectId: number, orderedIds: number[]) =>
    ipcRenderer.invoke('projectBullets:reorder', projectId, orderedIds),
},
```
