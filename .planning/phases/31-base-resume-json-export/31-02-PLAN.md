```ts
const isEmpty = (v: unknown): boolean =>
  v === null || v === undefined ||
  (typeof v === 'string' && v.trim() === '') ||
  (Array.isArray(v) && v.length === 0) ||
  (typeof v === 'object' && v !== null && !Array.isArray(v) && Object.keys(v).length === 0)

const opt = <K extends string, V>(key: K, value: V): Record<K, V> | Record<string, never> =>
  isEmpty(value) ? {} : ({ [key]: value } as Record<K, V>)
```

       ```ts
       /**
        * Builds the base resume.json document from the experience DB.
        *
        * Lossy-faithful contract — these fields are NOT exported:
        *   - profiles[1..]  (only first social profile kept)
        *   - basics.location.{address,postalCode,region,countryCode}  (only city kept)
        *   - basics.url, basics.label, basics.image
        *   - skills[].level
        *   - projects[].{description,url,startDate,endDate}
        *
        * Re-importing previously exported data creates duplicates — import is append-only.
        *
        * @throws ExportValidationError if produced object fails ResumeJsonSchema.parse
        */
       export function buildBaseResumeJson(db: Db): ResumeJson {
         // 1. Read all tables (mirror mergeHelper read order)
         const profileRow = db.select().from(profile).get()
         const jobRows = db.select().from(jobs).orderBy(asc(jobs.sortOrder)).all()
         const jobBulletRows = db.select().from(jobBullets).orderBy(asc(jobBullets.sortOrder)).all()
         // ... skills, projects, education, volunteer, awards, publications, languages, interests, referenceEntries
         //   (read each via db.select().from(<table>).all() with appropriate ordering)

         // 2. Build basics with conditional spread (D-04). Include profile.linkedin as profiles[0] only.
         const basicsLocation = {
           ...opt('city', profileRow?.location?.trim()),  // D-16: only city kept
         }
         const basics = {
           ...opt('name', profileRow?.name?.trim()),
           ...opt('email', profileRow?.email?.trim()),
           ...opt('phone', profileRow?.phone?.trim()),
           ...opt('summary', profileRow?.summary?.trim()),
           ...opt('location', basicsLocation),
           ...opt('profiles', profileRow?.linkedin?.trim()
             ? [{ url: profileRow.linkedin.trim() }]
             : undefined),
         }

         // 3. Build collections (filter empty entries)
         const work = jobRows
           .map((job) => toWorkEntry(job, jobBulletRows.filter((b) => b.jobId === job.id)))
           .filter((e) => Object.keys(e).length > 0)
         // ... same for education, skills, projects, awards, publications, volunteer, languages, interests, references

         // 4. Assemble top-level with opt() so empty groups are omitted
         const result: unknown = {
           ...opt('basics', basics),
           ...opt('work', work),
           ...opt('education', education),
           ...opt('skills', skillsOut),
           ...opt('projects', projectsOut),
           ...opt('awards', awardsOut),
           ...opt('publications', publicationsOut),
           ...opt('volunteer', volunteerOut),
           ...opt('languages', languagesOut),
           ...opt('interests', interestsOut),
           ...opt('references', referencesOut),
         }

         // 5. Validate-or-throw (D-02)
         const parsed = ResumeJsonSchema.safeParse(result)
         if (!parsed.success) {
           throw new ExportValidationError(parsed.error.issues)
         }
         return parsed.data
       }
       ```

| T-31-07 | Tampering | DB rows producing invalid JSON | mitigate | `ResumeJsonSchema.safeParse` rejects malformed output (D-02). `ExportValidationError` prevents Plan 03 from writing bad data to disk (D-15 — no save dialog on failure). |
| T-31-08 | Injection | DB string values containing JSON metacharacters (quotes, backslashes) | mitigate | `JSON.stringify` (Plan 03) handles escaping — Builder produces a plain JS object tree, no string concatenation. Verified safe via standard JSON serializer contract. |
| T-31-09 | Denial of Service | Very large DB → very large JSON object → memory/time blowup | accept | Single-user local app, resume DB is bounded (hundreds of bullets at most). No DoS surface. |
| T-31-10 | Spoofing | Builder consuming a Db handle not backed by the real schema | accept | Type system enforces `BetterSQLite3Database<typeof schema>` arg. Caller-controlled. |

No `high` severity threats. Validation-first ordering (D-13) means Plan 03 NEVER writes invalid data — the builder is the gate. This plan introduces the gate.
